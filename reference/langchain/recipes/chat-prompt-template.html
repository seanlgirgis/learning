Lab 20

ChatPromptTemplate + MessagesPlaceholder

Format with prompt.invoke, then prompt | llm.

ChatPromptTemplate

from langchain_core.prompts import ChatPromptTemplate
from langchain_helper import make_llm, GenParams

llama_llm = make_llm({GenParams.MAX_NEW_TOKENS: 120})

joke_prompt = ChatPromptTemplate.from_messages([
    ("system", "Helpful assistant. One response only."),
    ("user", "Tell me a joke about {topic}"),
])
joke_input = {"topic": "cats"}

print(joke_prompt.invoke(joke_input))
print((joke_prompt | llama_llm).invoke(joke_input))

MessagesPlaceholder

from langchain_core.messages import HumanMessage
from langchain_core.prompts import MessagesPlaceholder

history_prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    MessagesPlaceholder("msgs"),
])
history_input = {"msgs": [HumanMessage(content="What is the day after Tuesday?")]}

print((history_prompt | llama_llm).invoke(history_input))
