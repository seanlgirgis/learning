Lab 18 · Exercise 1 local

Compare temperature 0.8 vs 0.1

Two make_llm() instances — same model, different params.

Code

import os
from langchain_helper import make_llm, GenParams

print("Model:", os.environ["WATSONX_MODEL_ID"])

llm_creative = make_llm({GenParams.TEMPERATURE: 0.8})
llm_precise = make_llm({GenParams.TEMPERATURE: 0.1})

prompts = [
    "Write a short poem about artificial intelligence",
    "What are the key components of a neural network?",
    "List 5 tips for effective time management",
]

for prompt in prompts:
    print("PROMPT:", prompt)
    print("creative:", llm_creative.invoke(prompt))
    print("precise:", llm_precise.invoke(prompt))
    print()
