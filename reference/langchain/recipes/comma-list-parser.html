CommaSeparatedListOutputParser

Model returns comma-separated text → Python list.

Code

from langchain_classic.output_parsers import CommaSeparatedListOutputParser
from langchain_core.prompts import PromptTemplate
from langchain_helper import make_llm

list_parser = CommaSeparatedListOutputParser()
format_instructions = list_parser.get_format_instructions()

prompt = PromptTemplate(
    template="Answer the user query.\n{format_instructions}\nList five {subject}.",
    input_variables=["subject"],
    partial_variables={"format_instructions": format_instructions},
)

chain = prompt | make_llm() | list_parser
print(chain.invoke({"subject": "programming languages"}))
